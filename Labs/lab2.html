public class TowerOfHanoi {
public static void solveHanoi(int n, String start, String end, String middle){
if(n == 1)
System.out.println("Move 1 disk from " + start + " to " + end);
else{
solveHanoi(n-1, start, middle, end);
solveHanoi(1, start, end, middle);
solveHanoi(n-1, middle, end, start);
}
}
public static void main(String[] args){
solveHanoi(3, "A", "C", "B");
}
}
